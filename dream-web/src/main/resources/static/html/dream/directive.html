16进制常见的表示方法:
十进制值:100-->十六进制:0x64
十进制值:100-->十六进制:64h

加法指令用例 add

mov ax,8h
mov bx,4h
add ax,bx
------------
结果保存在ax中
(ax)= 000ch(12)

减法指令用例 sub

mov ax,8h
mov bx,4h
sub ax,bx
------------
结果保存在ax中
(ax)= 0004h(4)

乘法指令用例 mul

8位*8位
mov al,8h
mov bl,4h
mul bl          ;bl中保存一个乘数
------------
结果保存在ax中
(ax)= 0020h(32)

*********************************
16位*16位
mov ax,64h      ;100
mov bx,8h
mul bx          ;bx中保存一个乘数
------------
结果:800,低16位在ax中,高16位在dx中
(ax)= 0320h(800) (dx) = 0000f(0)

除法指令用例 div

8位/8位
mov al,9h
mov bl,2h
div bl          ;bl中保存除数
------------
商:4,保存在al中,余数:1,保存在ah中
(al)= 4h(4) (ah) = 1h(1)

*********************************
16位/16位
mov ax,d8773h   ;886643
mov bx,2h
div bx          ;bx中保存除数
------------
商:443321,保存在ax中,余数:1,保存在dx中
(ax)= 6c3b9h(443321) (dx) = 1h(1)
